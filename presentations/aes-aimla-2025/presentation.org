* Differentiation arithmetic

** Maybe don't use this                                            :noexport:

#+REVEAL: split data-transition="slide-in fade-out"

#+attr_html: :class huge-eqn
\begin{equation}
f'(x) = \lim_{h\to0}\frac{f(x + h) - f(x)}{h}
\end{equation}

#+REVEAL: split data-transition="fade"

#+attr_html: :class huge-eqn
\begin{equation}
f'(x) = \frac{f(x + \varepsilon\dot{x}) - f(x)}{\varepsilon\dot{x}}
\end{equation}

\(\dot{x}\) --- Newton's notation for the derivative.

\(\varepsilon\) --- a /nilpotent symbol/: \(\quad\varepsilon \neq
0\;,\quad\varepsilon^{2} = 0\)

#+REVEAL: split data-transition="fade"

Rearrange, and let \(f(x) = x\)

#+attr_html: :class huge-eqn
\begin{align}
f(x + \varepsilon\dot{x}) &= f(x) + \varepsilon\dot{x}f'(x) \\ 
  &= {\color{blue} x} + {\color{green} \varepsilon\dot{x}}
\end{align}

** Dual Number Arithmetic                                          :noexport:
:PROPERTIES:
:reveal_extra_attr: data-auto-animate
:END:

\begin{align}
(u + \varepsilon\dot{u}) + (v + \varepsilon\dot{v}) &= {\color{blue} u + v} + {\color{green} \varepsilon(\dot{u} + \dot{v})} \\[.5em]
(u + \varepsilon\dot{u}) - (v + \varepsilon\dot{v}) &= {\color{blue} u - v} + {\color{green} \varepsilon(\dot{u} - \dot{v})} \\[.5em]
(u + \varepsilon\dot{u}) \times (v + \varepsilon\dot{v}) &= {\color{blue} uv} + {\color{green} \varepsilon(u\dot{v} + v\dot{u})} \\[.5em]
(u + \varepsilon\dot{u}) \div (v + \varepsilon\dot{v}) &= {\color{blue} \frac{u}{v}} + {\color{green} \varepsilon\left(\frac{v\dot{u} - u\dot{v}}{v^{2}}\right)}
\end{align}

\begin{gather}
U+V = \langle {\color{blue} u}, {\color{green} u'} \rangle + \langle {\color{blue} v}, {\color{green} v'} \rangle
  = \langle {\color{blue} u+v}, {\color{green} u' + v'} \rangle \\
U-V = \langle {\color{blue} u}, {\color{green} u'} \rangle - \langle {\color{blue} v}, {\color{green} v'} \rangle
  = \langle {\color{blue} u+v}, {\color{green} u' - v'} \rangle \\
UV = \langle {\color{blue} u}, {\color{green} u'} \rangle\langle {\color{blue} v}, {\color{green} v'} \rangle
  = \langle {\color{blue} uv}, {\color{green} uv' + vu'} \rangle \\
\frac{U}{V} = \frac{\langle {\color{blue} u}, {\color{green} u'} \rangle}{\langle {\color{blue} v}, {\color{green} v'} \rangle}
  = \left\langle {\color{blue} \frac{u}{v}}, {\color{green} \frac{vu' - uv'}{v^{2}}} \right\rangle
\end{gather}

#+begin_src js
class Dual {
    constructor(primal, tangent) {
        this.primal = primal;
        this.tangent = tangent;
    }

    toString() {
        return `\\langle{\\color{blue}${this.primal.toFixed(2)}},{\\color{green}${this.tangent.toFixed(2)}}\\rangle`;
    }

    add(d) {
        return new Dual(this.primal + d.primal, this.tangent + d.tangent);
    }

    sub(d) {
        return new Dual(this.primal - d.primal, this.tangent - d.tangent);
    }

    mul(d) {
        return new Dual(this.primal * d.primal, this.primal * d.tangent + d.primal * this.tangent);
    }
};

                    CANVAS.PADDING + (yMax - f(new Dual(x, 1)).primal) * (AXES.Y.ZERO - CANVAS.PADDING) / yMax,
                    indicatorWidth,
                    0, 2 * Math.PI
                );
                ctx.fill();
            },

            draw(x) {
                this.drawTangent(x);
                this.drawIntersection(x);
            },

            redraw(x) {
                this.clear();
                this.draw(x);
            }
        }),

        async init() {
            if (this.canvas.getContext) {
                this.canvas.setAttribute("width", CANVAS.WIDTH);
                this.canvas.setAttribute("height", CANVAS.HEIGHT);

                return this.artist(this.canvas.getContext("2d"));
            }
        }
    }
};
