\begin{equation}
f'(x) = \lim_{h\to0}\frac{f(x + h) - f(x)}{h}
\end{equation}

#+REVEAL: split data-transition="fade"

#+attr_html: :class huge-eqn
\begin{equation}
f'(x) = \frac{f(x + \varepsilon\dot{x}) - f(x)}{\varepsilon\dot{x}}
\end{equation}

\(\dot{x}\) --- Newton's notation for the derivative.

\(\varepsilon\) --- a /nilpotent symbol/: \(\quad\varepsilon \neq
0\;,\quad\varepsilon^{2} = 0\)

#+REVEAL: split data-transition="fade"

#+attr_html: :class huge-eqn
\begin{equation}
\varepsilon\dot{x}f'(x) = f(x + \varepsilon\dot{x}) - f(x)
\end{equation}

#+REVEAL: split data-transition="fade"

#+attr_html: :class huge-eqn
\begin{equation}
f(x + \varepsilon\dot{x}) = f(x) + \varepsilon\dot{x}f'(x)
\end{equation}

#+REVEAL: split data-transition="fade"

\begin{equation}
\textrm{Let}\; f(x) = x
\end{equation}

#+attr_html: :class huge-eqn
\begin{align}
f(x + \varepsilon\dot{x}) &= f(x) + \varepsilon\dot{x}f'(x) \\ 
  &= {\color{blue} x} + {\color{green} \varepsilon\dot{x}}
\end{align}

const dynamicLayer = {
    canvas: document.getElementById("secantCanvas"),
    xSlider: {
        element: document.getElementById("xSquared-x"),
        valSpan: document.getElementById("xSquared-xValue")
    },
    hSlider: {
        element: document.getElementById("xSquared-h"),
        valSpan: document.getElementById("xSquared-hValue"),
        prevValue: 0
    },
    math: document.getElementById("xSquaredDynamicMath"),

    visualiser: ctx => ({
        drawIndicators: (x, h) => {
            const drawXIndicator = (x, y) => ctx.fillRect(
                CANVAS.PADDING - (AXES.X.INDICATOR.WIDTH / 2) + (x + X.MAX) * (AXES.WIDTH - CANVAS.PADDING * 2) / (X.MAX - X.MIN),
                y - AXES.X.INDICATOR.HEIGHT / 2,
                AXES.X.INDICATOR.WIDTH,
                AXES.X.INDICATOR.HEIGHT
            );

            const drawFxIndicator = (x, y) => {
                const indicatorWidth = 6;
                ctx.beginPath();
                ctx.arc(
                    CANVAS.PADDING + (x + X.MAX) * (AXES.WIDTH - CANVAS.PADDING * 2) / (X.MAX - X.MIN),
                    CANVAS.PADDING + (yMax - y) * (AXES.Y.ZERO - CANVAS.PADDING) / yMax,
                    indicatorWidth,
                    0, 2 * Math.PI
                );
                ctx.fill();
            };

            const drawSVG = (img, x, y, scale) => {
                const w = img.naturalWidth * scale;
                const h = img.naturalHeight * scale;
                const xPos = CANVAS.PADDING - w / 2 +
                      (x + X.MAX) * (AXES.WIDTH - CANVAS.PADDING * 2) / (X.MAX - X.MIN);
                const yPos = CANVAS.PADDING + (yMax - y) * (AXES.Y.ZERO - CANVAS.PADDING) / yMax;
                ctx.drawImage(img, xPos, yPos, w, h);
            };

            drawXIndicator(x, AXES.Y.ZERO);
            drawXIndicator(x + h, AXES.Y.ZERO);

            drawFxIndicator(x, f(x));
            drawFxIndicator(x + h, f(x + h));

            drawSVG(SVG.X, x, -.08, 1.5);
            drawSVG(SVG.XPLUSH, x + h, -.2, 1.5);
        },

        drawSecant: (x, h) => {
            const slope = (f(x + h) - f(x)) / h;
            const ySecVec = xVec.map(xVal => ({x: xVal, y: slope * (xVal - x) + f(x)}));
            ctx.beginPath();
            ctx.moveTo(
                CANVAS.PADDING + (ySecVec[0].x + X.MAX) * (AXES.WIDTH - CANVAS.PADDING * 2) / (X.MAX - X.MIN),
                CANVAS.PADDING + (yMax - ySecVec[0].y) * (AXES.Y.ZERO - CANVAS.PADDING) / yMax
            );
            ySecVec.forEach(p => ctx.lineTo(
                CANVAS.PADDING + (p.x + X.MAX) * (AXES.WIDTH - CANVAS.PADDING * 2) / (X.MAX - X.MIN),
                CANVAS.PADDING + (yMax - p.y) * (AXES.Y.ZERO - CANVAS.PADDING) / yMax
            ));
            ctx.strokeStyle = "green";
            ctx.lineWidth = 2;
            ctx.stroke();
        },

        clear: () => ctx.clearRect(0, 0, CANVAS.WIDTH, CANVAS.HEIGHT),
        
        draw(x, h) {
            this.drawIndicators(x, h);
            this.drawSecant(x, h);
        },
        
        redraw(x, h) {
            this.clear();
            this.draw(x, h)
        }
    }),

    async init() {
        if (this.canvas.getContext) {
            this.canvas.setAttribute("width", CANVAS.WIDTH);
            this.canvas.setAttribute("height", CANVAS.HEIGHT);
            
            let img = new Image();
            img.src = tex2imageSrc("x");
            await img.decode();
            SVG.X = img;
            
            img = new Image();
            img.src = tex2imageSrc("x + h");
            await img.decode();
            SVG.XPLUSH = img;

            const x = Number(this.xSlider.element.value);
            const h = Number(this.hSlider.element.value);                
            const v = this.visualiser(this.canvas.getContext("2d"));
            v.draw(x, h);

            const updateMath = async () => {
                const x = Number(this.xSlider.element.value);
                const h = Number(this.hSlider.element.value);
                const xx = f(x);
                const xhxh = f(x + h);
                const fprimex = (xhxh - xx) / h;
                const mx = fprimex / x;

                MathJax.typesetClear([this.math, this.xSlider.valSpan, this.hSlider.valSpan]);
                this.math.innerHTML = `$$
    \\begin{align}

#+begin_src js
class Dual {
    constructor(primal, tangent) {
        this.primal = primal;
        this.tangent = tangent;
    }

    toString() {
        return `\\langle{\\color{blue}${this.primal.toFixed(2)}},{\\color{green}${this.tangent.toFixed(2)}}\\rangle`;
    }

    add(d) {
        return new Dual(this.primal + d.primal, this.tangent + d.tangent);
    }

    sub(d) {
        return new Dual(this.primal - d.primal, this.tangent - d.tangent);
    }

    mul(d) {
        return new Dual(this.primal * d.primal, this.primal * d.tangent + d.primal * this.tangent);
    }
};
