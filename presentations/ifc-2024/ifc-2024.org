Some display constants:

#+begin_src js :results none :tangle src/xsquared.js
const CANVAS = {
    WIDTH: 600,
    HEIGHT: 600,
    PADDING: 25
};

const AXES = {
    X: {
        ZERO: CANVAS.WIDTH / 2,
        INDICATOR: {
            WIDTH: 3,
            HEIGHT: 10
        }
    },
    Y: {
        ZERO: CANVAS.HEIGHT - CANVAS.PADDING - 50,
    }
};

const X = {
    MIN: -2,
    MAX: 2,
    LEN: 1000
};

X.STEP = (X.MAX-X.MIN)/X.LEN;
#+end_src

A vector for the independent variable \(x\):

#+begin_src js :results none :tangle src/xsquared.js
const xVec = Array.from({length: X.LEN + 1}, (_,x) => X.MIN + x * X.STEP);
#+end_src

Prepare to plot \(f(x) = x^{2}\):

#+begin_src js :results none :tangle src/xsquared.js
export default async () => {
    const yVec = xVec.map(x => ({x: x, y: x * x}));
    const yMax = Math.max(...yVec.map(y => y.y));
    const f = x => x * x;

    // Cache the NPM version of MathJax, otherwise it'll be overwritten by the version bundled by Reveal.js
    const mj = MathJax;
    await mj.startup.promise;
    const xml = new XMLSerializer();
    const tex2imageSrc = tex => "data:image/svg+xml;base64, " +
          window.btoa(unescape(encodeURIComponent(
              xml.serializeToString(mj.tex2svg(tex).firstElementChild)
          )));
#+end_src

Define the axes and the static plot of \(f(x)\):

#+begin_src js :results none :tangle src/xsquared.js
    const staticLayer = {
        canvas: document.getElementById("xSquaredCanvas"),

        visualiser: ctx => ({
            drawAxis: {
                x: () => {
                    ctx.beginPath();
                    ctx.moveTo(CANVAS.PADDING, AXES.Y.ZERO);
                    ctx.lineTo(CANVAS.WIDTH - CANVAS.PADDING, AXES.Y.ZERO);
                    ctx.stroke();
                    // Arrow
                    ctx.beginPath();
                    ctx.moveTo(CANVAS.WIDTH - CANVAS.PADDING, AXES.Y.ZERO);
                    ctx.lineTo(CANVAS.WIDTH - CANVAS.PADDING - 10, AXES.Y.ZERO - 5);
                    ctx.lineTo(CANVAS.WIDTH - CANVAS.PADDING - 10, AXES.Y.ZERO + 5);
                    ctx.closePath();
                    ctx.fill();
                },
                y: () => {
                    ctx.beginPath();
                    ctx.moveTo(AXES.X.ZERO, CANVAS.PADDING);
                    ctx.lineTo(AXES.X.ZERO, CANVAS.HEIGHT - CANVAS.PADDING);
                    ctx.stroke();
                    // Arrow
                    ctx.beginPath();
                    ctx.moveTo(AXES.X.ZERO, CANVAS.PADDING);
                    ctx.lineTo(AXES.X.ZERO - 5, CANVAS.PADDING + 10);
                    ctx.lineTo(AXES.X.ZERO + 5, CANVAS.PADDING + 10);
                    ctx.closePath();
                    ctx.fill();
                    // Label
                    ctx.drawImage(SVG.FX, AXES.X.ZERO - 60, CANVAS.PADDING,
                                 SVG.FX.naturalWidth * 1.5,
                                 SVG.FX.naturalHeight * 1.5);
                }
            },

            drawFunction: () => {
                ctx.beginPath();
                ctx.moveTo(
                    CANVAS.PADDING + (yVec[0].x + X.MAX) * (CANVAS.WIDTH - CANVAS.PADDING * 2) / (X.MAX - X.MIN),
                    CANVAS.PADDING + (yMax - yVec[0].y) * (AXES.Y.ZERO - CANVAS.PADDING) / yMax
                );
                yVec.forEach(p => ctx.lineTo(
                    CANVAS.PADDING + (p.x + X.MAX) * (CANVAS.WIDTH - CANVAS.PADDING * 2) / (X.MAX - X.MIN),
                    CANVAS.PADDING + (yMax - p.y) * (AXES.Y.ZERO - CANVAS.PADDING) / yMax
                ));
                ctx.stroke();
            },
        }),

        async init() {
            if (this.canvas.getContext) {
                this.canvas.setAttribute("width", CANVAS.WIDTH);
                this.canvas.setAttribute("height", CANVAS.HEIGHT);

                let img = new Image();
                img.src = tex2imageSrc("f(x)");
                await img.decode();
                SVG.FX = img;

                const v = this.visualiser(this.canvas.getContext("2d"));
                v.drawAxis.x();
                v.drawAxis.y();
                v.drawFunction();
            }
        }
    };

Define the dynamic part of the plot --- the secant and axis
indicators. Slope \(m\) found as \(m = \frac{f(x + h) - f(x)}{h}\);
secant line found as \(m(x - x_{1}) + y_{1}\).

#+begin_src js :results none :tangle src/xsquared.js
    const dynamicLayer = {
        canvas: document.getElementById("secantCanvas"),
        xSlider: {
            element: document.getElementById("xSquared-x"),
            valSpan: document.getElementById("xSquared-xValue")
        },
        hSlider: {
            element: document.getElementById("xSquared-h"),
            valSpan: document.getElementById("xSquared-hValue"),
            prevValue: 0
        },
        math: document.getElementById("xSquaredDynamicMath"),

        visualiser: ctx => ({
            drawIndicators: (x, h) => {
                const drawXIndicator = (x, y) => ctx.fillRect(
                    CANVAS.PADDING - (AXES.X.INDICATOR.WIDTH / 2) + (x + X.MAX) * (CANVAS.WIDTH - CANVAS.PADDING * 2) / (X.MAX - X.MIN),
                    y - AXES.X.INDICATOR.HEIGHT / 2,
                    AXES.X.INDICATOR.WIDTH,
                    AXES.X.INDICATOR.HEIGHT
                );

                const drawFxIndicator = (x, y) => {
                    const indicatorWidth = 6;
                    ctx.beginPath();
                    ctx.arc(
                        CANVAS.PADDING + (x + X.MAX) * (CANVAS.WIDTH - CANVAS.PADDING * 2) / (X.MAX - X.MIN),
                        CANVAS.PADDING + (yMax - y) * (AXES.Y.ZERO - CANVAS.PADDING) / yMax,
                        indicatorWidth,
                        0, 2 * Math.PI
                    );
                    ctx.fill();
                };

                const drawSVG = (img, x, y, scale) => {
                    const w = img.naturalWidth * scale;
                    const h = img.naturalHeight * scale;
                    const xPos = CANVAS.PADDING - w / 2 +
                          (x + X.MAX) * (CANVAS.WIDTH - CANVAS.PADDING * 2) / (X.MAX - X.MIN);
                    const yPos = CANVAS.PADDING + (yMax - y) * (AXES.Y.ZERO - CANVAS.PADDING) / yMax;
                    ctx.drawImage(img, xPos, yPos, w, h);
                };

                drawXIndicator(x, AXES.Y.ZERO);
                drawXIndicator(x + h, AXES.Y.ZERO);

                drawFxIndicator(x, f(x));
                drawFxIndicator(x + h, f(x + h));

                drawSVG(SVG.X, x, -.08, 1.5);
                drawSVG(SVG.XPLUSH, x + h, -.2, 1.5);
            },

            drawSecant: (x, h) => {
                const slope = (f(x + h) - f(x)) / h;
                const ySecVec = xVec.map(xVal => ({x: xVal, y: slope * (xVal - x) + f(x)}));
                ctx.beginPath();
                ctx.moveTo(
                    CANVAS.PADDING + (ySecVec[0].x + X.MAX) * (CANVAS.WIDTH - CANVAS.PADDING * 2) / (X.MAX - X.MIN),
                    CANVAS.PADDING + (yMax - ySecVec[0].y) * (AXES.Y.ZERO - CANVAS.PADDING) / yMax
                );
                ySecVec.forEach(p => ctx.lineTo(
                    CANVAS.PADDING + (p.x + X.MAX) * (CANVAS.WIDTH - CANVAS.PADDING * 2) / (X.MAX - X.MIN),
                    CANVAS.PADDING + (yMax - p.y) * (AXES.Y.ZERO - CANVAS.PADDING) / yMax
                ));
                ctx.stroke();
            },

            clear: () => ctx.clearRect(0, 0, CANVAS.WIDTH, CANVAS.HEIGHT),
            
            draw(x, h) {
                this.drawIndicators(x, h);
                this.drawSecant(x, h);
            },
            
            redraw(x, h) {
                this.clear();
                this.draw(x, h)
            }
        }),

        async init() {
            if (this.canvas.getContext) {
                this.canvas.setAttribute("width", CANVAS.WIDTH);
                this.canvas.setAttribute("height", CANVAS.HEIGHT);
                
                let img = new Image();
                img.src = tex2imageSrc("x");
                await img.decode();
                SVG.X = img;
                
                img = new Image();
                img.src = tex2imageSrc("x + h");
                await img.decode();
                SVG.XPLUSH = img;

                const x = Number(this.xSlider.element.value);
                const h = Number(this.hSlider.element.value);
                const xx = f(x);
                const xhxh = f(x + h);
                const fprimex = (xhxh - xx) / h;
                const mx = fprimex / x;
                
                const v = this.visualiser(this.canvas.getContext("2d"));
                v.draw(x, h);

                const updateMath = async () => {
                    const x = Number(this.xSlider.element.value);
                    const h = Number(this.hSlider.element.value);
                    const xx = f(x);
                    const xhxh = f(x + h);
                    const fprimex = (xhxh - xx) / h;
                    const mx = fprimex / x;

                    mj.typesetClear([this.math, this.xSlider.valSpan, this.hSlider.valSpan]);
                    this.math.innerHTML = `$$
    \\begin{align}
        f'(x) &\\approx \\frac{f(x + h) - f(x)}{h} \\\\[10pt]
          &\\approx \\frac{${xhxh.toFixed(2)} - ${xx.toFixed(2)}}{${h}} = ${fprimex.toFixed(2)} \\\\[10pt]
          &\\approx ${isFinite(mx) ? mx.toFixed(2) : "\\infty{}"}x
    \\end{align}
    $$`;
                    this.xSlider.valSpan.innerHTML = `\\(${x.toFixed(2)}\\)`;
                    this.hSlider.valSpan.innerHTML = `\\(${h.toFixed(2)}\\)`;
                    await mj.typesetPromise([this.math, this.xSlider.valSpan, this.hSlider.valSpan]);
                };

                updateMath();

                this.xSlider.element && this.xSlider.valSpan &&
                    this.xSlider.element.addEventListener("input", async ({target: {value: val}}) => {
                        v.redraw(Number(this.xSlider.element.value), Number(this.hSlider.element.value));
                        await updateMath();
                    });

                this.hSlider.element && this.hSlider.valSpan &&
                    this.hSlider.element.addEventListener("input", async ({target: {value: val}}) => {
                        // Avoid division by zero!
                        if (val == 0) {
                            val = this.hSlider.prevValue > val ? -.01 : .01;
                            this.hSlider.element.value = val;
                        }
                        this.hSlider.prevValue = val;
                        v.redraw(Number(this.xSlider.element.value), Number(this.hSlider.element.value));
                        await updateMath();
                    });
            }
        }
    };
#+end_src
