const CANVAS = {
    WIDTH: 550,
    HEIGHT: 550,
    PADDING: 10
};

const AXES = {
    X: {
	ZERO: CANVAS.WIDTH / 2,
    },
    Y: {
	ZERO: CANVAS.HEIGHT - CANVAS.PADDING - 50,
    }
};

const X = {
    MIN: -5,
    MAX: 5,
    LEN: 1000
};

X.STEP = (X.MAX-X.MIN)/X.LEN;

const xVec = Array.from({length: X.LEN + 1}, (_,x) => X.MIN + x * X.STEP);

const xSquared = () => {
    const canvas = document.getElementById("xSquared");
    const xSlider = document.getElementById("xSquared-x");
    const xValue = document.getElementById("xSquared-xValue");
    const hSlider = document.getElementById("xSquared-h");
    const hValue = document.getElementById("xSquared-hValue");
    
    if (canvas.getContext) {
        canvas.setAttribute("width", CANVAS.WIDTH);
        canvas.setAttribute("height", CANVAS.HEIGHT);
	const ctx = canvas.getContext("2d");
        
        const yVec = xVec.map(x => ({x: x, y: x * x}));
        const yMax = Math.max(...yVec.map(y => y.y));

	// Draw axes
	ctx.beginPath();
	ctx.moveTo(CANVAS.PADDING, AXES.Y.ZERO);
	ctx.lineTo(CANVAS.WIDTH - CANVAS.PADDING, AXES.Y.ZERO);
	ctx.stroke();

	ctx.beginPath();
	ctx.moveTo(AXES.X.ZERO, CANVAS.PADDING);
	ctx.lineTo(AXES.X.ZERO, CANVAS.HEIGHT - CANVAS.PADDING);
	ctx.stroke();

	// Draw function
        ctx.beginPath();
        ctx.moveTo(
	    CANVAS.PADDING + (yVec[0].x + X.MAX) * (CANVAS.WIDTH - CANVAS.PADDING * 2) / (X.MAX - X.MIN),
            CANVAS.PADDING + (yMax - yVec[0].y) * (AXES.Y.ZERO - CANVAS.PADDING) / yMax
	);
        yVec.forEach(p => ctx.lineTo(
            CANVAS.PADDING + (p.x + X.MAX) * (CANVAS.WIDTH - CANVAS.PADDING * 2) / (X.MAX - X.MIN),
            CANVAS.PADDING + (yMax - p.y) * (AXES.Y.ZERO - CANVAS.PADDING) / yMax
	));
        ctx.stroke();

	// Draw secant
	const xVal = Number(xSlider.value);
	const hVal = Number(hSlider.value);
	const ySec = ((xVal + hVal) * (xVal + hVal) - xVal * xVal) / hVal;
	const ySecVec = xVec.map(x => ({x: x + xVal + hVal, y: ySec * x}));
	ctx.beginPath();
        ctx.moveTo(
	    CANVAS.PADDING + (ySecVec[0].x + X.MAX) * (CANVAS.WIDTH - CANVAS.PADDING * 2) / (X.MAX - X.MIN),
            CANVAS.PADDING + (yMax - ySecVec[0].y) * (AXES.Y.ZERO - CANVAS.PADDING) / yMax
	);
        ySecVec.forEach(p => ctx.lineTo(
            CANVAS.PADDING + (p.x + X.MAX) * (CANVAS.WIDTH - CANVAS.PADDING * 2) / (X.MAX - X.MIN),
            CANVAS.PADDING + (yMax - p.y) * (AXES.Y.ZERO - CANVAS.PADDING) / yMax)
        );
        ctx.stroke();
    }

    xSlider && xValue && xSlider.addEventListener("input", ({target: {value: v}}) => {
	xValue.textContent = v;
	// Redraw secant
    }); 
    hSlider && hValue && hSlider.addEventListener("input", ({target: {value: v}}) => {
	hValue.textContent = v;
	// Redraw secant
    });
}
