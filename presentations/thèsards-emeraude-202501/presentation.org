#+begin_src julia :results none
function plotDrift(dataCategory, plotFilepath)
    t41lastBytes = [191, 203, 234]
    plot(legendtitle="IP address", yformatter=:scientific, size=(850, 475), margin=5Plots.mm)
    xlabel!("Time (s)")
    ylabel!("Offset (s)");
    for b in t41lastBytes
        column = data[!, "$b-$dataCategory"]
        # Remove initial offset
        column .= column .- column[1]
        plot!(column, label="...48.$b", linewidth=1.5)
    end

    savefig(plotFilepath)
end
#+end_src

** Sample-synchronous (almost) networked audio

#+ATTR_REVEAL: :frag (appear)
- Four Teensies
  #+ATTR_REVEAL: :frag (appear)
  + One acts as an audio server --- transmits a 1 Hz impulse train to
    the network.
    #+ATTR_REVEAL: :frag (appear)
    * Also acts as a /clock authority/ --- transmits PTP sync messages
      to the network.
  + Three are audio clients --- they receive, buffer and reproduce the
    impulse train signal.
- Packets are timestamped
  + Clients select the packet with timestamp closest to their current
    ethernet clock time.
